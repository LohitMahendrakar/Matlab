%% 1. Trapezoidal Numerical Integration
x_trapz = 1;
y_trapz = [0 4 5 3 10 11 2];
Q = trapz(x_trapz, y_trapz);

%% 2. Eigenvalues and Eigenvectors
A_eig = [2 2; 1 3];
[v, d] = eig(A_eig); 

%% 3. Linear Algebraic Equations
A_linear = [2,  1,  2; 
           -1,  1, -1; 
           -1,  2,  3];

B_linear = [2; 3; -10];
x_linear = linsolve(A_linear, B_linear);
disp('Linear System Solution:');
disp(x_linear);

%% 4. RK Method (ode45)
tspan = [1 1.05];
y0 = 1.2;
[t_ode, y_ode] = ode45(@(t, y) t.^2 + y.^2, tspan, y0);
disp('ODE45 Results [t, y]:');
disp([t_ode, y_ode]);

%% 5. Application: Static Problem (Resultant Force)
n = 3;
alpha = [40 25 58];
alpha1 = alpha * pi / 180; % Convert degrees to radians
force = [20 40 200];
sx = 0;
sy = 0;

for i = 1:n
    sx = sx + force(i) * cos(alpha1(i));
    sy = sy + force(i) * sin(alpha1(i));
end

r = sqrt(sx^2 + sy^2);
ar = atan(sy / sx);

if ar < 0
    ar = ar + 360; % Note: If ar is in radians, adding 360 (degrees) can cause bugs. 
end                % Consider using atan2d(sy, sx) for direct degree calculation.

disp('Static Problem Resultant (r) and Angle (ar):');
disp(r);
disp(ar); 

%% 6. Curve Fitting
x_fit = [0 2 4 5 9];
y_fit = [1.1 2.9 4.7 5.1 8];
cl = polyfit(x_fit, y_fit, 1);
disp('Curve Fitting Coefficients (Slope and Intercept):');
disp(cl);
